208struct DatabaseWindow
209{
210 std::vector<Poco::Dynamic::Var> cache;
211 int row1, column1;
212 int row2, column2;
213
214 DatabaseWindow() : row1(0), column1(0), row2(0), column2(0) {}
215
216 bool is_data_found_in_window(int row, int column) const
217 {
218 return row >= row1 && row < row2 && column >= column1 && column < column2;
219 }
220
221 void resize_window(int r1, int c1, int r2, int c2)
222 {
223 row1 = std::min(r1, r2);
224 column1 = std::min(c1, c2);
225 row2 = std::max(r1, r2);
226 column2 = std::max(c1, c2);
227 cache.resize((row2 - row1) * (column2 - column1));
228 }
229
230 void clear()
231 {
232 this->resize_window(0, 0, 0, 0);
233 }
234};

386 // Use the already validated and sanitized table name directly

406inline void DatabaseMatrix::count_columns()
407{
408 try
409 {
410 column_names_.clear();
411
412 // Clear the cache
413 cache_window_.clear();
414
415 // Construct and execute the SQL query with a WHERE condition
416 Poco::Data::Statement column_statement(session_);
417 column_statement << "SELECT * FROM " << table_name_.get();
418
419 if (!condition_.empty())
420 column_statement << " WHERE " << condition_;
421
422 column_statement << " LIMIT 1";
423 column_statement.execute();
424
425 // Retrieve column names from the result metadata
426 Poco::Data::RecordSet record_set(column_statement);
427 for (size_t i = 0; i < record_set.columnCount(); ++i)
428 {
429 column_names_.push_back(record_set.columnName(i));
430 }
431
432 if (column_names_.empty())
433 {
434 last_error_ = "No columns found for table " + table_name_.get();
435 }
436 }
437 catch (const std::exception& e)

465inline DatabaseMatrix::value_type DatabaseMatrix::at_(int64_t row, int64_t column) const
466{
467 // Check if the data is in the current cache window
468 if (!cache_window_.is_data_found_in_window(row, column))
469 {
470 // If not, preload the data around the requested row and column
471 preload_data(row, column);
472 }
473
474 // Calculate the index in the cache for the requested data
475 size_t cache_index = (row - cache_window_.row1) * (cache_window_.column2 - cache_window_.column1) + (column - cache_window_.column1);
476
477 // Return the data from the cache
478 return cache_window_.cache[cache_index];
479}

485inline void DatabaseMatrix::preload_data(int64_t row, int64_t column) const
486{
487 // Determine the range of rows to preload
488 int64_t start_row = std::max(int64_t(0), row - int64_t(cache_window_size_) / int64_t(2));
489 int64_t end_row = std::min(start_row + int64_t(cache_window_size_), int64_t(this->rows()));
490
491 // Resize and reset the cache window
492 cache_window_.resize_window(start_row, 0, end_row, this->columns());
493
494 // Construct and execute the SQL query
495 Poco::Data::Statement select(session_);
496
497 select << "SELECT * FROM " << table_name_.get();
498
499 if (!condition_.empty())
500 select << " WHERE " << condition_;
501
502 if (!row_sorting_method_.get().empty())
503 select << " ORDER BY " << row_sorting_method_.get();
504
505 select << " LIMIT " << (end_row - start_row) << " OFFSET " << start_row;
506
507 select.execute();
508
509 // Load the data into the cache
510 Poco::Data::RecordSet record_set(select);
511 size_t cache_index = 0;
512 bool more = record_set.moveFirst();
513 while (more)
514 {
515 for (size_t i = 0; i < record_set.columnCount(); ++i)
516 {
517 cache_window_.cache[cache_index++] = record_set[i];
518 }
519 more = record_set.moveNext();
520 }
521}
