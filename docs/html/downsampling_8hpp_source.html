73 typename MatrixType2,

77inline void simple_downsampling(const MatrixType1& source,
78 MatrixType2& destination,
79 int64_t start_index,
80 int64_t end_index,
81 bool sample_rows)
82{
83 // Determine the step, referring to how many source
84 // values we skip as we downsample
85 double step = double(end_index - start_index) / double(sample_rows ? destination.columns() : destination.rows());
86
87 // Iterate over each row if downsampling rows or column otherwise
88 for(int64_t source_i = 0, dest_i = 0;
89 source_i < (sample_rows ? source.rows() : source.columns()) &&
90 dest_i < (sample_rows ? destination.rows() : destination.columns());
91 ++source_i, ++dest_i)
92 {
93 int64_t dest_index = 0;
94 double source_index = double(start_index);
95
96 // Now for each row (or column), we need to downsample the source
97 for(;
98 dest_index < (sample_rows ? destination.columns() : destination.rows()) &&
99 ((step > 0) ? (source_index < end_index) : (source_index > end_index));
100 ++dest_index, source_index += step)
101 {
102 if(sample_rows)
103 destination(dest_i, dest_index) = source.circ_at(source_i, int64_t(source_index));
104 else
105 destination(dest_index, dest_i) = source.circ_at(int64_t(source_index), source_i);
106 }
107 }
108}

137 typename MatrixType2,

141inline void downsample_lttb(const MatrixType1& source,
142 MatrixType2& destination,
143 int64_t source_start_index,
144 int64_t source_end_index)
145{
146 // Calculate the absolute size of the source data to sample
147 size_t source_size = std::abs(source_end_index - source_start_index);
148 size_t destination_size = destination.size();
149
150 // Check if downsampling is needed
151 if (destination_size == 0 || source_size == 0)
152 return;
153
154 // Directly copy source to destination if destination size is larger or equal to source
155 if (destination_size >= source_size)
156 {
157 size_t count = 0;
158 for (int64_t i = source_start_index; count < source_size; ++i, ++count)
159 {
160 destination(count) = source.circ_at(i);
161 }
162 return;
163 }
164
165 // If destination size is 1, just take the first point from the source
166 if (destination_size == 1)
167 {
168 destination(0) = source.circ_at(source_start_index);
169 return;
170 }
171
172 // Calculate the number of data points to skip (bucket size)
173 auto every = static_cast<double>(source_size - 2) / static_cast<double>(destination_size - 2);
174
175 int64_t a_index = source_start_index;
176 destination(0) = source.circ_at(a_index); // Always keep the first point
177 size_t dest_index = 1; // Start filling the destination from the second element
178
179 // Iterate over each bucket to find the best point
180 for (size_t i = 0; i < destination_size - 2; ++i)
181 {
182 // Calculate the average point for the next bucket (containing c)
183 double avg_y = 0;
184 int64_t avg_range_start = source_start_index + static_cast<int64_t>((i + 1) * every) + 1;
185 int64_t avg_range_end = source_start_index + static_cast<int64_t>((i + 2) * every) + 1;
186 int64_t avg_range_length = std::abs(avg_range_end - avg_range_start);
187 for (int64_t idx = avg_range_start; idx < avg_range_end; ++idx)
188 {
189 avg_y += source.circ_at(idx);
190 }
191 avg_y /= avg_range_length;
192 double avg_x = (avg_range_start + avg_range_end) / 2.0; // Average x-value for bucket
193
194 // Find the point that forms the largest triangle
195 double max_area = -1;
196 int64_t next_a_index = a_index;
197
198 for (int64_t range_offs = avg_range_start; range_offs < avg_range_end; ++range_offs)
199 {
200 // Calculate the area of the triangle formed by points a, b, and avg
201 double x0 = a_index, y0 = source.circ_at(a_index);
202 double x1 = range_offs, y1 = source.circ_at(range_offs);
203 double area = std::abs((x0 - avg_x) * (y1 - y0) - (x0 - x1) * (avg_y - y0)) / 2;
204
205 // Select the point that maximizes the area (forming the largest triangle)
206 if (area > max_area)
207 {
208 max_area = area;
209 next_a_index = range_offs;
210 }
211 }
212
213 // Add the selected point to the destination at the current index
214 destination(dest_index++) = source.circ_at(next_a_index);
215 a_index = next_a_index; // Update the starting point for the next triangle
216 }
217
218 // Always include the last point from the source
219 destination(dest_index) = source.circ_at(source_end_index);
220}

241template<typename MatrixType1,
242 typename MatrixType2,

250inline void downsample_lttb_xy(const MatrixType1& x_source,
251 const MatrixType2& y_source,
252 MatrixType3& x_destination,
253 MatrixType4& y_destination,
254 int64_t source_start_index,
255 int64_t source_end_index)
256{
257 size_t source_size = std::abs(source_end_index - source_start_index);
258 size_t destination_size = x_destination.size(); // Assume x and y destinations have the same size
259
260 if (destination_size == 0 || source_size == 0)
261 return;
262
263 if (destination_size >= source_size)
264 {
265 size_t count = 0;
266 for (int64_t i = source_start_index; count < source_size; ++i, ++count)
267 {
268 x_destination(count) = x_source.circ_at(i);
269 y_destination(count) = y_source.circ_at(i);
270 }
271 return;
272 }
273
274 if (destination_size == 1)
275 {
276 x_destination(0) = x_source.circ_at(source_start_index);
277 y_destination(0) = y_source.circ_at(source_start_index);
278 return;
279 }
280
281 auto every = static_cast<double>(source_size - 2) / static_cast<double>(destination_size - 2);
282
283 int64_t a_index = source_start_index;
284 x_destination(0) = x_source.circ_at(a_index);
285 y_destination(0) = y_source.circ_at(a_index);
286 size_t dest_index = 1;
287
288 for (size_t i = 0; i < destination_size - 2; ++i)
289 {
290 double avg_x = 0, avg_y = 0;
291 int64_t avg_range_start = source_start_index + static_cast<int64_t>((i + 1) * every) + 1;
292 int64_t avg_range_end = source_start_index + static_cast<int64_t>((i + 2) * every) + 1;
293 int64_t avg_range_length = std::abs(avg_range_end - avg_range_start);
294 for (int64_t idx = avg_range_start; idx < avg_range_end; ++idx)
295 {
296 avg_x += x_source.circ_at(idx);
297 avg_y += y_source.circ_at(idx);
298 }
299 avg_x /= avg_range_length;
300 avg_y /= avg_range_length;
301
302 double max_area = -1;
303 int64_t next_a_index = a_index;
304
305 for (int64_t range_offs = avg_range_start; range_offs < avg_range_end; ++range_offs)
306 {
307 double x0 = x_source.circ_at(a_index), y0 = y_source.circ_at(a_index);
308 double x1 = x_source.circ_at(range_offs), y1 = y_source.circ_at(range_offs);
309 double area = std::abs((x0 - avg_x) * (y1 - y0) - (x0 - x1) * (avg_y - y0)) / 2;
310
311 if (area > max_area)
312 {
313 max_area = area;
314 next_a_index = range_offs;
315 }
316 }
317
318 x_destination(dest_index) = x_source.circ_at(next_a_index);
319 y_destination(dest_index++) = y_source.circ_at(next_a_index);
320 a_index = next_a_index;
321 }
322
323 x_destination(dest_index) = x_source.circ_at(source_end_index);
324 y_destination(dest_index) = y_source.circ_at(source_end_index);
325}
