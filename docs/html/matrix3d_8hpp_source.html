49// Header and Footer section of a multi-dimentional memory mapped matrix
50//-------------------------------------------------------------------
51const std::string matrix3d_header_byte_sequence = "---begin_3d_---\n";
52const std::string matrix3d_footer_byte_sequence = ":---end_3d_---:\n";
53
54struct Matrix3DHeader
55{
56 char header[16] = {'-', '-', '-', 'b', 'e', 'g', 'i', 'n', '_', '3', 'd', '_', '-', '-', '-', '\n'};
57 std::mutex lock;
58 uintptr_t size_of_data_type = 8;
59
60 uintptr_t pages = 0;
61 uintptr_t rows = 0;
62 uintptr_t columns = 0;
63};
64
65struct Matrix3DFooter
66{
67 char footer[16] = {':', '-', '-', '-', 'e', 'n', 'd', '_', '3', 'd', '_', '-', '-', '-', ':', '\n'};

73//-------------------------------------------------------------------
74// Function used to check whether a memory section does indeed
75// contain a memory mapped matrix
76//-------------------------------------------------------------------
77inline bool does_memory_contain_mapped_matrix3d(const char* memory_mapped_matrix3d,
78 uintptr_t memory_size_in_bytes)
79{
80 uintptr_t minimum_size = sizeof(Matrix3DHeader) + sizeof(Matrix3DFooter);
81
82 if(memory_size_in_bytes < minimum_size)
83 return false;
84
85 const Matrix3DHeader* header = reinterpret_cast<const Matrix3DHeader*>(memory_mapped_matrix3d);
86
87 uintptr_t expected_size = minimum_size + header->size_of_data_type * header->rows * header->columns;
88
89 return memory_size_in_bytes >= expected_size;
90}

391template<typename DataType>
392
393inline std::error_code Matrix3D<DataType>::create_matrix(uintptr_t pages,
394 uintptr_t rows,
395 uintptr_t columns,
396 const DataType& initial_value,
397 const std::string& matrix_filename)
398{
399 std::error_code mapping_error;
400
401 // Check first if the current map is not
402 // already the correct size
403 if(mapped_file_.is_open())
404 {
405 if(this->capacity() >= pages * rows * columns)
406 {
407 // This means the matrix we are currently
408 // mapped to is of equal or greater size
409 // than the matrix we need, so we just re-initialize
410 // and fit the new matrix within this one
411 // No need to create a new file and map to it
412
413 if(pages * rows * columns == 0)
414 {
415 this->get_header()->pages = 0;
416 this->get_header()->rows = 0;
417 this->get_header()->columns = 0;
418 }
419 else
420 {
421 this->get_header()->pages = pages;
422 this->get_header()->rows = rows;
423 this->get_header()->columns = columns;
424
425 this->initialize(initial_value);
426 }
427
428 return mapping_error;
429 }
430 }
431
432 mapped_file_.unmap();
433
434 filename_of_memory_mapped_file_ = matrix_filename;
435
436 // If the user specified a filename, we attempt
437 // to create the matrix using that filename, otherwise
438 // we create the matrix using a unique name in the temp
439 // directory
440 if(filename_of_memory_mapped_file_.empty())
441 {
442 // The user did not specify a filename, so
443 // we get a temporary filename
444 filename_of_memory_mapped_file_ = std::tmpnam(nullptr);
445 }
446
447 // Calculate the necessary size of the file
448 // so that it can hold the matrix
449 uintptr_t size_of_file = sizeof(Matrix3DHeader) + sizeof(Matrix3DFooter) + pages*rows*columns*sizeof(DataType);
450
451 // Create the file
452 std::fstream file_to_create(filename_of_memory_mapped_file_,std::fstream::in | std::fstream::out | std::fstream::trunc);
453
454 if(!file_to_create.is_open())
455 {
456 mapping_error.assign(1,std::iostream_category());
457 return mapping_error;
458 }
459
460 file_to_create.close();
461
462 // Resize the file to the correct size
463 fs::resize_file(filename_of_memory_mapped_file_, size_of_file, mapping_error);
464
465 if(mapping_error)
466 return mapping_error;
467
468 // Finally we memory map the entire file
469 mapped_file_.map(filename_of_memory_mapped_file_, mapping_error);
470
471 if(mapping_error)
472 return mapping_error;
473
474 // Write the Matrix Header
475 std::copy(matrix3d_header_byte_sequence.cbegin(),
476 matrix3d_header_byte_sequence.cend(),
477 &this->get_header()->header[0]);
478 this->get_header()->size_of_data_type = sizeof(DataType);
479 this->get_header()->pages = pages;
480 this->get_header()->rows = rows;
481 this->get_header()->columns = columns;
482
483 // Write the Matrix Footer
484 std::copy(matrix3d_footer_byte_sequence.cbegin(),
485 matrix3d_footer_byte_sequence.cend(),
486 &this->get_footer()->footer[0]);
487
488 // Finally we initialize the matrix values
489 this->initialize(initial_value);
490
491 // We are done
492 return mapping_error;
493}

498//-------------------------------------------------------------------
499// Function used to load a matrix from file
500//-------------------------------------------------------------------
501template<typename DataType>
502
503inline std::error_code Matrix3D<DataType>::load_matrix(const std::string& file_to_load_matrix_from)
504{
505 std::error_code mapping_error;
506
507 mapped_file_.unmap();
508 filename_of_memory_mapped_file_= file_to_load_matrix_from;
509
510 // First we check if the file exists
511 if(!fs::exists(filename_of_memory_mapped_file_))
512 {
513 // The file does not exist, therefore, we
514 // return an error
515 mapping_error.assign(1,std::iostream_category());
516 return mapping_error;
517 }
518
519 // We now know the file exists, so we check its size
520 // to make sure it is sized correctly to host the
521 // matrix it supposedly hosts
522 uintptr_t minimum_file_size_needed_to_hold_a_matrix = sizeof(Matrix3DHeader) + sizeof(Matrix3DFooter);
523
524 if(fs::file_size(filename_of_memory_mapped_file_) < minimum_file_size_needed_to_hold_a_matrix)
525 {
526 // The specified file is smaller than the minimum
527 // size required to hold matrix information, so
528 // we return an error
529 mapping_error.assign(1,std::iostream_category());
530 return mapping_error;
531 }
532
533 // Finally we memory map the entire file
534 mapped_file_.map(filename_of_memory_mapped_file_, mapping_error);
535
536 if(mapping_error)
537 return mapping_error;
538
539 // Now that we have mapped the file, we check
540 // whether the file is actually storing a mapped
541 // matrix
542 if(!does_memory_contain_mapped_matrix3d(mapped_file_.data(), mapped_file_.size()))
543 {
544 // This means the file did not actually store
545 // a mapped matrix, so we just return an error
546 mapping_error.assign(1,std::iostream_category());
547 return mapping_error;
548 }
549
550 // We are done
551 return mapping_error;
552}
