56template<typename MatrixType1,
57 typename MatrixType2>
58
59struct ElementByElementBinaryExpression : public BaseMatrix< ElementByElementBinaryExpression<MatrixType1, MatrixType2> >
60{
61 // Type of value that is stored in left side expression
62 // - We use that type to return the same type for the
63 // resulting binary operation (i.e. m1 + m2 returns type of m1(0,0))
64 using value_type = typename std::remove_reference<decltype(std::declval<MatrixType1>()(0,0))>::type;
65
66 // The operation function type
67 using operation_type = std::function<value_type(value_type, value_type)>;
68
69
70
71 ElementByElementBinaryExpression<MatrixType1, MatrixType2>(const MatrixType1& left_side_expression,
72 const MatrixType2& right_side_expression,
73 const operation_type& operation_function)
74 : left_side_expression_(left_side_expression),
75 right_side_expression_(right_side_expression),
76 operation_function_(operation_function)
77 {
78 }
79
80
81
82 // This operation assumes both expressions are the
83 // same size
84 uintptr_t rows()const
85 {
86 return this->left_side_expression_.rows();
87 }
88
89 uintptr_t columns()const
90 {
91 return this->left_side_expression_.columns();
92 }
93
94
95
96 const MatrixType1& get_left_side_expression()
97 {
98 return left_side_expression_;
99 }
100
101 const MatrixType2& get_right_side_expression()
102 {
103 return right_side_expression_;
104 }
105
106
107
108 decltype(auto) at_(int row, int column)const
109 {
110 return static_cast<value_type>(operation_function_(this->left_side_expression_.at(row,column),
111 this->right_side_expression_.at(row,column)));
112 }
113
114
115
116private:
117
118 const MatrixType1& left_side_expression_;
119 const MatrixType2& right_side_expression_;
120 operation_type operation_function_;
121};
