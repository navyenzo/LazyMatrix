227 // Advances the iterator by a specified number of steps
228 CircularIterator<ContainerType>& advance(std::ptrdiff_t movement)
229 {
230 if(current_pointer_ == end_pointer_)
231 return (*this);
232
233 current_position_accumulative_ += movement;
234 current_number_of_rotations_ = current_position_accumulative_ / this->size();
235
236 if(current_number_of_rotations_ >= maximum_number_of_rotations_ || current_number_of_rotations_ <= -maximum_number_of_rotations_)
237 {
238 current_pointer_ = end_pointer_;
239 return (*this);
240 }
241
242 std::ptrdiff_t desired_position_in_container = (this->size() + current_position_accumulative_ % this->size()) % this->size();
243
244 if(std::abs(desired_position_in_container - current_position_in_container_) <= (this->size() / 2))
245 {
246 std::advance(current_pointer_, desired_position_in_container - current_position_in_container_);
247 }
248 else if(desired_position_in_container < current_position_in_container_)
249 {
250 current_pointer_ = begin_pointer_;

252 }
253 else
254 {
255 current_pointer_ = end_pointer_;
256 std::advance(current_pointer_, desired_position_in_container - this->size());
257 }

260
