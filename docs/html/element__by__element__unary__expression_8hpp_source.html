58{
59 // Type of value that is stored in expression
60 // - We use that type to return the same type for the
61 // resulting unary operation (i.e. abs(m1(0,0)) returns type of m1(0,0))
62 using value_type = typename std::remove_reference<decltype(std::declval<MatrixType>()(0,0))>::type;
63
64 // The operation function type
