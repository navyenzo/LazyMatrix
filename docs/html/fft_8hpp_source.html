63inline void fft_moving_window(const Eigen::Matrix<Scalar, Eigen::Dynamic, Eigen::Dynamic>& data, // Input matrix of data
64 Eigen::Matrix<std::complex<Scalar>, Eigen::Dynamic, Eigen::Dynamic>& result, // Output matrix of FFT results
65 const double window_duration_sec, // Duration of the time window in seconds
66 const double sampling_period_sec, // Sampling period or frequency of the data in seconds
67 const int num_overlap_steps) // Number of overlap steps between successive windows
68{
69 // Create an FFT object for computing the FFT
70 Eigen::FFT<typename Eigen::Matrix<Scalar, Eigen::Dynamic, 1>::Index> fft;
71
72 // Compute the size of the time window and the number of steps for the moving FFT
73 const int window_size = static_cast<int>(window_duration_sec / sampling_period_sec);
74 const int num_channels = data.rows();
75 const int num_samples = data.cols();
76 const int num_steps = (num_samples - window_size) / (window_size - num_overlap_steps) + 1;
77
78 // Resize the output matrix to hold the FFT results
79 result.resize(num_channels, window_size / 2 + 1);
80
81 // Loop over each channel in the input data
82 for (int i = 0; i < num_channels; ++i)
83 {
84 // Loop over each time window in the data
85 for (int j = 0; j < num_steps; ++j)
86 {
87 // Extract the current window of data from the input matrix
88 Eigen::Matrix<Scalar, Eigen::Dynamic, 1> window = data.block(i, j * (window_size - num_overlap_steps), 1, window_size).transpose();
89
90 // Compute the FFT of the window
91 auto fft_result = fft.fwd(window);
92
93 // Copy the non-redundant half of the FFT output to the output matrix
94 for (int k = 0; k < window_size / 2 + 1; ++k)
95 {
96 result(i, k) = fft_result(k);
97 }
98 }
99 }
100}
